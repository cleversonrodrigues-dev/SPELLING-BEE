    >

    <button class="blue" onclick="pronounceWord()">
      Pronunciation
    </button>

    <button class="yellow" onclick="spellWord()">
      Spelling
    </button>

  </div>

  <script>

    function pronounceWord(){

      const word = document.getElementById("word").value;

      const speech = new SpeechSynthesisUtterance(word);

      speech.lang = "en-US";

      speechSynthesis.speak(speech);

    }

    function spellWord(){

      const word = document.getElementById("word").value;

      const spelled = word.split("").join(" ");

      const speech = new SpeechSynthesisUtterance(spelled);

      speech.lang = "en-US";

      speech.rate = 0.5;

      speechSynthesis.speak(speech);

    }

  </script>
